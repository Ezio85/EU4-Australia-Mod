		required_missions = {  }
		provinces_to_highlight = {
		}
		trigger = {
			is_year = 1445
			army_size_percentage = 1
			manpower_percentage = 0.6
			num_of_generals = 1
		}
		effect = {
			china_superregion = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
		}
	}
	unify_north_china = {
		icon = mission_great_wall
		required_missions = { prepare_for_unification }
		provinces_to_highlight = {
			region = north_china_region
		}
		trigger = {
			north_china_region = {
				type = all

					is_empty = yes
				}
			}
			
		}
		effect = {
			add_prestige = 25
			add_adm_power = 150

		}
	}
	unite_china = {
		icon = mission_unite_home_region
		required_missions = { unify_south_china unify_north_china unify_xinan }
		
		provinces_to_highlight = {
			region = north_china_region
			region = south_china_region
			region = xinan_region

			NOT = { owned_by = ROOT }
		}
		
		
		trigger = {
			china_superregion= {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
				NOT = {
					is_empty = yes
				}
			}
		}

		effect = {
			add_mil_power = 100
			add_adm_power = 100
			add_dip_power = 100
			add_prestige = 25
			add_country_modifier = {
				name = unifier_of_china
				duration = -1
			}
			manchuria_region = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
			if = {
				limit = {
					exists = KOR
				}
				add_casus_belli = {
					target = KOR
					type = cb_vassalize_mission
				}
			}
			else = {
				korea_region = {
					limit = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
					add_permanent_claim = ROOT
				}
			}
			tibet_region = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
		}
	}
	
	
	pacify_manchuria = {
		icon = mission_manchu_warrior
		required_missions = { unite_china }
		
		provinces_to_highlight = {
			region = manchuria_region
			NOT = { owned_by = ROOT }
			NOT = { area = sakhalin_area }
		}
		
		trigger = {
			manchuria_region = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
				NOT = {
					is_empty = yes
					unrest = 1
					area = sakhalin_area
				}
				dip_power = 200
			}
			
			
		}

		effect = {
			mongolia_region = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
			add_dip_power = -200
			custom_tooltip = manchu_culture_tt

			every_province = {
				limit = {
					culture = manchu
				}
				change_culture = manchu_new
			}
			every_country = {
				limit = {
					primary_culture = manchu
				}
				change_primary_culture = manchu_new
				set_ruler_culture = manchu_new
				if = {
					limit = {
						has_heir = yes
					}
					set_heir_culture = manchu_new
				}
				if = {
					limit = {
						has_consort = yes
						consort_culture = manchu
					}
					set_consort_culture = manchu_new
				}
			}
		}
	}
	
	end_the_hordes = {
		icon = mission_non-western_cavalry_raid
		required_missions = { pacify_manchuria }
		
		provinces_to_highlight = {
			region = mongolia_region
			NOT = { owned_by = ROOT }
		}
		
		trigger = {
			mongolia_region = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
				NOT = {
					is_empty = yes
				}
			}
			
			
		}

		effect = {
			every_owned_province = {
				limit = {
					culture_group = altaic
					region = mongolia_region
				}
				add_devastation = 50
			}
			add_treasury = 1000
			add_adm_power = 100
			add_mil_power = 100
			add_dip_power = 100
			shanshan_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
			
			kashgaria_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
			tarim_basin_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
			
		}
	}
	
	establish_xinjiang = {
		icon = mission_steppe_warriors
		required_missions = { end_the_hordes }
		
		provinces_to_highlight = {
			area = shanshan_area
			area = kashgaria_area
			area = tarim_basin_area
			NOT = { owned_by = ROOT }
		}
		
		trigger = {

			}
			kashgaria_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
			tarim_basin_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT

			}
			
			
		}

		effect = {
			every_owned_province = {
				limit = {
					culture = uyghur
				}
				add_province_modifier = {
					name = happy_culture_exchange
					duration = 3650
				}
				add_devastation	= 50
			}
		}
	}
}
