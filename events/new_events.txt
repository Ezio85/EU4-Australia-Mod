namespace = new_events

#St. Domingue Event
country_event = {
	id = new_events.1
	title = new_events.1.t
	desc = new_events.1.d
	picture = TRADEGOODS_eventPicture
	fire_only_once = yes

	major = yes

	trigger = {
		NOT = { has_global_flag = sugar_capital_happened }
		any_owned_province = {
				region = carribeans_region
				trade_goods = sugar
				has_building = plantations
				development = 15
			}
	}

	mean_time_to_happen = {
		months = 12
	}
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					region = carribeans_region
					trade_goods = sugar
					has_building = plantations
					development = 15
				}
				save_event_target_as = sugar_capital
			}
		}
	}
	option = {
		name = new_events.1.a
		event_target:sugar_capital = {
			add_permanent_province_modifier = {
				name = pearl_of_the_antilles
				duration = -1
			}
		}
		set_global_flag =  sugar_capital_happened
		}
}

#Startup Event
#+1 Stab at Start of game
#Transports for England
#Redistributes Electorships to players
#Stop Prussia Cheesing

country_event = {

	id = new_events.4
	title = "new_events_4.t"
	desc = "new_events_4.d"
	picture = TRADEGOODS_eventPicture

	fire_only_once = yes

	is_triggered_only = yes

	hidden = yes

	trigger = {
		owns = 1
	}

	immediate = {
		every_country = {
			add_yearly_manpower = 3
			add_stability = 1
		}
		#### +4 Transports for England
		236 = {
			ENG = {
				transport = PREV
			}
		}
		236 = {
			ENG = {
				transport = PREV
			}
		}
		236 = {
			ENG = {
				transport = PREV
			}
		}
		236 = {
			ENG = {
				transport = PREV
			}
		}
		####Makes Every player in the HRE an elector
		every_country = {
			limit = {
				ai = no
				is_part_of_hre = yes
				NOT = { is_emperor = yes }
				NOT = { is_elector = yes }
			}
			elector = yes
		}
		###Independence for Holland if player
		HOL = {
			if = {
				limit = {
					ai = no
				}
				grant_independence = yes
			}

		}
		####Grants spare elector slots to AI
		if = {
			limit = {
				NOT = { num_of_electors = 7 }
			}
			SAX = {
				elector = yes
			}
		}
		if = {
			limit = {
				NOT = { num_of_electors = 7 }
			}
			TRI = {
				elector = yes
			}
		}
		if = {
			limit = {
				NOT = { num_of_electors = 7 }
			}
			PAL = {
				elector = yes
			}
		}
		if = {
			limit = {
				NOT = { num_of_electors = 7 }
			}
			KOL = {
				elector = yes
			}
		}
		if = {
			limit = {
				NOT = { num_of_electors = 7 }
			}
			MAI = {
				elector = yes
			}
		}
		###Gives country flag to all eligible Prussia candidates
		every_country = {
			limit = {
				OR = {
					tag = LIV
					tag = SAX
					tag = STE
					tag = WOL 
					tag = MKL 
					tag = ANH
					tag = MAG
					tag = MKL
					tag = RIG
					tag = THU
				}
			}
			set_country_flag = can_form_prussia_flag
		}
		

	}

	option = {
		name = free_stab_event
	}
}
